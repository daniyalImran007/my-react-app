name: CI Pipeline

# === TRIGGERS ===
on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]
  workflow_dispatch:  # Manual trigger from GitHub UI

# === CONCURRENCY: Prevent overlapping runs on same branch/PR ===
concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

# === PERMISSIONS (Security best practice) ===
permissions:
  contents: read
  actions: read
  security-events: write  # Required for CodeQL
  statuses: write         # For commit status

jobs:
  build-and-test:
    name: Build & Test (Node ${{ matrix.node }})
    runs-on: ubuntu-latest

    # === MATRIX STRATEGY: Test across multiple Node.js versions ===
    strategy:
      matrix:
        node-version: [18.x, 20.x, 22.x]
      fail-fast: false  # Continue even if one fails

    steps:
      # --- 1. Checkout Code ---
      - name: Checkout Repository
        uses: actions/checkout@v4
        with:
          fetch-depth: 0  # Full history for better analysis

      # --- 2. Setup Node.js ---
      - name: Setup Node.js ${{ matrix.node-version }}
        uses: actions/setup-node@v4
        with:
          node-version: ${{ matrix.node-version }}

      # --- 3. CACHING DEPENDENCIES (node_modules + npm cache) ---
      - name: Cache npm dependencies
        uses: actions/cache@v4
        id: npm-cache
        with:
          path: |
            ~/.npm
            node_modules
          key: ${{ runner.os }}-npm-${{ hashFiles('package-lock.json') }}
          restore-keys: |
            ${{ runner.os }}-npm-

      # --- 4. Install Dependencies ---
      - name: Install dependencies
        run: npm ci

      # --- 5. Lint Code ---
      - name: Run ESLint
        run: npm run lint
        continue-on-error: false

      # --- 6. Run Unit Tests ---
      - name: Run Tests
        run: npm test -- --coverage --watchAll=false

      # --- 7. Build Production Bundle ---
      - name: Build React App
        run: npm run build

      # --- 8. UPLOAD ARTIFACT: Production build ---
      - name: Upload Build Artifact
        uses: actions/upload-artifact@v4
        with:
          name: react-build-node-${{ matrix.node-version }}
          path: build/
          retention-days: 7

  # === CODE SCANNING / SECURITY (CodeQL) ===
  codeql:
    needs: build-and-test  # Only run if build passes
    runs-on: ubuntu-latest
    permissions:
      security-events: write
      actions: read
      contents: read

    steps:
      - name: Checkout Code
        uses: actions/checkout@v4

      - name: Initialize CodeQL
        uses: github/codeql-action/init@v3
        with:
          languages: javascript-typescript
          queries: security-and-quality

      - name: Perform CodeQL Analysis
        uses: github/codeql-action/analyze@v3

  # === NOTIFICATIONS: Slack on Failure ===
  notify:
    needs: [build-and-test, codeql]
    runs-on: ubuntu-latest
    if: failure() && github.ref == 'refs/heads/main'
    steps:
      - name: Send Slack Notification
        uses: slackapi/slack-github-action@v1.26.0
        with:
          payload: |
            {
              "text": ":fire: *CI Pipeline Failed!* `${{ github.workflow }}` on `${{ github.ref_name }}`\n<${{ github.server_url }}/${{ github.repository }}/actions/runs/${{ github.run_id }}|View Workflow Run>",
              "channel": "#ci-alerts"
            }
        env:
          SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK_URL }}
          SLACK_WEBHOOK_TYPE: INCOMING_WEBHOOK
